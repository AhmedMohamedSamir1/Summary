<< React Hooks >>

- useState() // to rerender the component 
	let [stateName, setStateName] = use(init val to state)
	let [count, setCount] = useState(0)
	// count = 0;
	// setCount is the function that will be called to change the value(state) of count
	// commonly method call the method that change the state of variable setCount(count+1)

- useEffect()
	- called explicitly after initial render of component 
	- useEffect(callback function, array)
	- if array = [] // called only once, (component did mount)
	- if array = [count] // called  after initial rendering and called after any change in state count (component did update & mount)
	- if array not passed to useEffect the callback function will be called after init rendering and after any change of any state

<< consume API >>

-- npm i axios

import axios from "axios";

const baseURL = "http://localhost:3333/products";

const getAllProducts = () => axios.get(baseURL);

const getProductById = (productId) => axios.get(`${baseURL}/${productId}`);

const addNewProduct = (product) => axios.post(baseURL, product);

const editProduct = (productId, product) => axios.put(`${baseURL}/${productId}`, product);

const deleteProduct = (productId) => axios.delete(`${baseURL}/${productId}`);

export {
	getAllProducts,
	getProductById,
	addNewProduct,
	editProduct,
	deleteProduct,
};
