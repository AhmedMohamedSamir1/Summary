- data-bs-toggle 
	- Applies to Button or link
	- tells Bootstrap what kind of component to open or toggle.
	- When clicked, Bootstrap should toggle (open)
	- EX: <button data-bs-toggle="modal" data-bs-target="#myModal"></button>

- data-bs-target
	- Applies to Button or link
	-  Specifies the target element to toggle or open
	- EX: data-bs-target="#myModal"

- data-bs-backdrop
	- Applies on Modal element
	- controls the behavior of the overlay behind the modal, which is the semi-transparent overlay that appears behind a modal dialog.
        - It can take three possible values
		- data-bs-backdrop="true"   // default, creates a backdrop that is visible and, when clicked, will close the modal.
		- data-bs-backdrop="static" // creates a backdrop, but clicking on it will not close the modal.
		- data-bs-backdrop="false"  // there will be no overlay behind the modal.

- data-bs-dismiss
	- Applies on Inside a component (element that lives inside the Bootstrap component that you want to close)
	- Tells Bootstrap to close the component
	- data-bs-dismiss="modal"
	- 

- data-bs-keyboard
	- Applies on  Modal element
	- Determines if ESC key can close the modal
	- EX: data-bs-keyboard="false"

- class="modal"
	- A Bootstrap class that makes the element modal

- class="fade"
	- fade: Adds a fade-in and fade-out transition effect when the modal is opened or closed.

- data-bs-dismiss="modal"
	- When the element with this attribute is clicked, it triggers the modal to close (i.e., it hides the modal from view).

- tabindex="-1"
	- Prevents the modal itself from being focused via the Tab key.

window.click = function(button) {

    console.log(button); // Get the tag name and ID from the button

    const tagName = button.getAttribute('data-name'); // Tag name
    const link = button.getAttribute('formaction');   // /Tag/Delete/25?pageNumber=1
    console.log(tagName);
    console.log(link);

    // Update the modal message with the tag name dynamically

    const deleteMessage = document.getElementById('deleteMessage');
    deleteMessage.innerHTML = `Are you sure you want to delete this Tag: <strong>${tagName}</strong>?`;

    const deleteLink = document.getElementById('deleteConfirmationSubmit');
    deleteLink.setAttribute('href', link);
 };
