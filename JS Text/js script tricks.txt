<< if your links or buttons one of them should have a specific class let's say "active" and other not when click on one of them class "active" added to it and removed from the others>>

        <div class="btn-group">
             <button type="button" class="btn btn-outline-primary">Left</button>
             <button type="button" class="btn btn-outline-primary">Middle</button>
             <button type="button" class="btn btn-outline-primary">Right</button>
        </div>

	<script>
            btn = document.querySelectorAll('div div button');
            btn.forEach(element => {
                console.log(element);
                element.addEventListener("click",()=>{
                    btn.forEach(element2 => {
                        element2.classList.remove('active');                          
                    });
                    element.classList.add('active');
                });
            });
        </script>
