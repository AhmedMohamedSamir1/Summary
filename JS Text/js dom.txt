<< events >>

1. add event at html element
	    <button onclick="display()">click here</button>

2. add event at js file using attach function
    	<button id="my-btn">click here</button>
    	let  myBtn = document.getElementById('my-btn');
    	myBtn.onclick = function(){ console.log('hello ahmed');}

    	<p id = 'mainP'>hello from p</p>
    	let myP = document.getElementById("mainP");
    	function one(){ console.log('message from onClick event ');}
    	myP.onclick = one;   // the ordinary event when you click on paragraph message from onClick event will be printed in console

3. add event at js file using add event listener 
    <p id = 'mainP'>hello from p</p>
    let myP = document.getElementById("mainP");
    myP.addEventListener("click", function(){console.log('message from addEventListener click 11');});

     // event listener allows you to execute more than one function at one envent
     myP.addEventListener("click", function(){console.log('message from addEventListener click 22');});
     the output on click on the paragraph : message from addEventListener click 11
				   	    message from addEventListener click 22

4. very important note: 
       - when event fired it sends an event object as a parameter to the function that will be executed, 
	 this object represent info about element that fire the event 

	<p id = 'mainP'>hello from p</p>
	myP.addEventListener("click", test);
	function test(e){
    		console.log(e.target);     // <p id = 'mainP'>hello from p</p>
    		console.log("test says hello");
	}
	// <button onclick="test(event)">click here</button> //event is a keyword that represent event object sent to the function

5. remove event listener
	 myP.removeEventListener("click", test); 
	// you may have more than action method in one event, here the method test will not be executed when click is occured

 
6. the submit of form refreshes the page so to prevent this behavior
	myBtn.onclick = function(event){ event.preventDefault(); }
	
7. the behaviour of event is :
	- when event of element is fired, the engine of js will ask the element parent if it has the same event if true it will fire also its event then 
 	  go to it parent of parent and ask if it has the same event if true it will fire it and so on until reach the body, this proccess called propagation
	- to stop propagation
	  myBtn.onclick = function(event){ event.stopPropagation(); }

- simulated events
	window.onload = function(){ document.links[0].click(); }

<< localStorage object >>

- stored in browser storage
- 

// set 
	window.localStorage.setItem("color","red");
	localStorage.fontSize="30px";
	localStorage["fontWeight"]="bold";

// get
	console.log(window.localStorage.getItem("color","red"));
	console.log(localStorage.fontSize);
	console.log(localStorage['fontWeight']);

// key
	console.log(localStorage.key(0)); // red
	console.log(localStorage.key(1)); // 30px

// localStorage length
	console.log(localStorage.length);

// removeItem
	localStorage.removeItem('fontWeight');

// clear 
	localStorage.clear();
