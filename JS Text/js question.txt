find the longest word of array using reduce function

let arr = ['ahmed', 'mohamed', 'samir', 'encyclopedia'];

console.log(arr.reduce(function(acc , curVal){

    return acc.length>curVal.length?acc:curVal;

}));

let str = "1,2,3,EE,l,z,r,o,_,W,e,b,_,S,cc,h,o,o,l,2,0,7"; // Elzero Web School; using split , reduce , filter , map

console.log(str.split(',').filter(
    (ele)=> isNaN(ele)
).map(
    (ele)=> ele.length>1?ele[0]:ele
).reduce(
    (accum, curval) => curval==='_'?accum+' ': accum+curval
