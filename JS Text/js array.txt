- map()
	- creates a new array from calling a function for every array element.
	- does not execute the function for empty elements.
	- does not change the original array.
	- can take a function as paramter
	- map(function(arrElement[req], currentElementIndex[opt], current_array[opt] array that map perform operation on){}, 
		thisArg[opt] An object to which the this keyword can refer to);
	- can takes a normal function as paramter

	let arr = [1,2,3,4];
	console.log(arr.map(element=>2*element)); // [2,4,6,8]
	console.log(arr.map(function(element){
    		return 5*element;
	}));   // [5,10,15,20]

	function double(num){
		return num*2;
	}
	console.log(arr.map(double));  // [2,4,6,8]

// < array destructing >>
// allow us to extract data from arrays, maps and objects and set them into distinct variables 

let arr = ["ahmed", "mohamed", "samir", "elsayed"];

let [a, b, c] = arr;
console.log(a); // ahmed
console.log(b); // mohamed
console.log(c); // samir

[a, , , b] = arr;
console.log(a); // ahmed
console.log(b); //elsayed


var arr = ["ahmed", "mohamed", "samir"];
var [y,x] = arr;
console.log(x); // mohamed
console.log(y); // ahmed

[y,,x] = arr;
	console.log(x); // samir
	console.log(y); // ahmed

[x,...y] = arr;
console.log(x);  // ahmed
console.log(y);  // ["mohamed", "samir"]


let person = {
	fname: "ali",
	lname: "ahmed",
	age: 20,
	color: "red",
};

let { fname: n, age: a } = person;
	console.log(n);  // ali
    	console.log(a);  // 20

let { fname: fname, age: lname } = person;  // the syntax sugar => let { fname, lname } = person; in case of key and variable is the same name
	console.log(n);  // ali
	console.log(a);  // ahmed
