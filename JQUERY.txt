<< actions >>

- each() : 
    $("h3").each(function(){
        console.log($(this).text());
    });  // it will get the text of each h3

- click : deprecated
    $("h3").click(function(){console.log("clicked")});

- on("event", function(){})
    $("h3").on("click", function(){console.log("clicked")});

-what if i want to make an click event to p elements that are created and will be created
    $("p").on("click", function(){}); // it will work with the created p only, what about p that will be created
    the sol: $(".container").on("click", "p", function(){}); //the current and future p inside .container will have click event

off() : to remove function from event 
    $(".container").off("click","p",functioName) // must assign functionName not anonymous function

- css()  // add properties in inline style
    $("h3").css("color", "red");
    $("h3").css({
        'padding':'10px',
        'background-color':"grey"
    }); // note do not write [;] inside ''

- addClass() , removeClass(), toggleClass()
    $("h3").eq(0).addClass("styledClass");

- attr()  // assign and get attribute of element
    $("h3").eq(0).attr("id", "hID");
        $("h3").attr({
            'id':'hID',
            'class':"myClass"
    });
    $("#w3s").attr("href"); // get attribute value

- append()
    let ele1 = $("<h3>hello from h3</h3>");
    let ele2 = $("<h3>hello from h3</h3>");
    $("h3").append(ele1, ele2);

- prepend() // insert at first
    $("h3").prepend(ele1);
    let item = $(".container").eq(0);
    $("body").prepend(item)  // it will move the element

- insertBefore() , insertAfter()
    let ele1 = $("<h3>hello from h3</h3>");
    $(ele1).insertBefore(".main");
    $(ele1).insertAfter(".main");

- before(), after()
    $(".main").before(ele1); // before .main insert ele1
    $(".main").after(ele1); // after   .main insert ele1

- remove()
    $("h3").eq(0).remove();

- empty()
    $("div").empty();

- clone()
    $("body").append($(".main").clone());
    let e = $("div").clone();

- hover()
    $("div h3").hover(mouseinFunc, mouseoutFunct)

<< animation >>

- slideUp() : it make element display: none
    $(".main").on("click",function(){
        // $(this).slideUp(1000);
        $(this).slideUp(2000 , function(){console.log("click")}); // the function will be fired when the event finished

    })

- slideDown() : it makes the element display from none to its default 

- slideToggle();
    $(".main").slideToggle(500);

- fadeOut() , fadeIn(), fadeToggle()

- animate(): customized animation
    $(".main").animate(
        {
            'width':'500px',
            'height':'500px',
        }
        ,2000)

stop(): 
     $("#animatedDiv2").stop(clearQueue,jumptoEnd);
     // clearQueue: animation clearQueue
     // jumptoEnd : move to the end of animation
     $("#animatedDiv2").stop(false,false);  // default
     $("#animatedDiv2").stop(true,false);   
     $("#animatedDiv2").stop(true,true);
     $("#animatedDiv2").stop(false,true);
